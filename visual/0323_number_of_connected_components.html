def countComponents(n, edges):
    """
    Union-Find with path compression.
    Time: O(E * Œ±(N)), Space: O(N)
    """
    parent = list(range(n))
    rank = [1] * n
    
    def find(x):
        if parent[x] != x:
            parent[x] = find(parent[x])  # Path compression
        return parent[x]
    
    def union(x, y):
        px, py = find(x), find(y)
        if px == py:
            return 0  # Already connected
        if rank[px] < rank[py]:
            px, py = py, px
        parent[py] = px
        rank[px] += rank[py]
        return 1  # Merged two components
    
    components = n
    for x, y in edges:
        components -= union(x, y)
    
    return components
