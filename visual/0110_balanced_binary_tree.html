#0110 Balanced Binary Tree

Problem: Determine if a binary tree is height-balanced. A height-balanced tree has subtrees that differ in height by at most 1.

Pattern: DFS - Check height difference at each node, return -1 if unbalanced

def isBalanced(root):
    """
    DFS returning height or -1 if unbalanced.
    Time: O(n), Space: O(h)
    """
    def dfs(node):
        if not node:
            return 0
        
        left = dfs(node.left)
        if left == -1:
            return -1
        
        right = dfs(node.right)
        if right == -1:
            return -1
        
        if abs(left - right) > 1:
            return -1
        
        return 1 + max(left, right)
    
    return dfs(root) != -1
