def gameOfLife(self, board):
    # States: 0=dead, 1=alive, 2=alive‚Üídead, 3=dead‚Üíalive
    m, n = len(board), len(board[0])
    
    for i in range(m):
        for j in range(n):
            neighbors = self.countNeighbors(board, i, j)
            
            if board[i][j] == 1:  # Currently alive
                if neighbors < 2 or neighbors > 3:
                    board[i][j] = 2  # Will die
            else:  # Currently dead
                if neighbors == 3:
                    board[i][j] = 3  # Will be born
    
    # Finalize states
    for i in range(m):
        for j in range(n):
            board[i][j] %= 2
