def missingNumber(nums):
    """
    XOR all indices (0 to n) with all values.
    Pairs cancel out, leaving the missing number.
    Time: O(n), Space: O(1)
    """
    n = len(nums)
    result = n  # Start with n (last index)
    
    for i in range(n):
        result ^= i ^ nums[i]
    
    return result
